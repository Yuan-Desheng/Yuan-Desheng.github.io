<script>
import {DICT_TYPE, getDictDatas} from "@/utils/dict";
import {mapGetters} from "vuex";

export default {
  name: 'SearchBar',
  props: {
    fatherQueryParams: {
      type: Object,
      default() {
        return {}
      }
    },
    fatherColumns: {
      type: Array,
      default() {
        return []
      }
    },
  },
  data() {
    return {
      queryParams: {},
      columns: [],
      searchTerm: [],
    }
  },

  computed: {
    ...mapGetters(['visible_column']),
  },

  watch: {
    searchTerm(val) {
      // if (this.searchTerm.length) {
        this.columns.forEach(columnItem => {
          columnItem.show = false;
        })
        val.forEach(item => {
          this.columns.forEach(columnItem => {
            if (columnItem.prop === item) {
              columnItem.show = true;
            }
          })
        })
      // }
    },
    columns: {
      deep: true,
      handler(val) {

        val.forEach(item => {
          this.queryParams[item.prop] = item.value;
        })
      }
    },
  },

  created() {
    this.initData();
  },
  methods: {
    // 初始化数据
    initData() {
      this.searchTerm = [];
      this.queryParams = _.cloneDeep(this.fatherQueryParams);
      this.columns = _.cloneDeep(this.fatherColumns);

      this.columns = this.columns.filter(item => this.visible_column.some(obj => obj.prop === item.prop));

      this.columns.forEach(item => {
        item.show = true;
        if (item.type === 'select') {
          if (item.dictType) {
            item.dictType = getDictDatas(DICT_TYPE[item.dictType]);
          } else {
            item.dictType = getDictDatas(item.prop);
          }
        } else if (item.type === 'multipleLikeInput') {
          let itemLikeProp = item.prop + 'Like';
          item.prop = itemLikeProp;
        }
      });
      this.columns.forEach(item => {
        this.searchTerm.push(item.prop);
      });
    },
    // 重置表单
    onEmpty() {
      this.initData();
      this.$emit('reprovision');
    },
    // 搜索
    onSearch() {
      this.$emit('search', this.queryParams)
    }
  }

}
</script>

v-drawerDrag 弹窗拖拽 ​

js
 /**
 * v-drawerDrag 弹窗拖拽
 * Copyright (c) 2019 ruoyi
 */
export default {
  bind(el, binding, vnode, oldVnode) {
    const value = binding.value
    if (value === false) return
    // 获取拖拽内容头部
    const drawerHeaderEl = el.querySelector('.el-drawer__header');
    const dragDom = el.querySelector('.el-drawer');
    drawerHeaderEl.style.cursor = 'move';
    // 获取原有属性 ie dom元素.currentStyle 火狐谷歌 window.getComputedStyle(dom元素, null);
    const sty = dragDom.currentStyle || window.getComputedStyle(dragDom, null);
    dragDom.style.position = 'absolute';
    dragDom.style.marginTop = 0;
    let width = dragDom.style.width;
    if (width.includes('%')) {
      width = +document.body.clientWidth * (+width.replace(/%/g, '') / 100);
    } else {
      width = +width.replace(/\px/g, '');
    }
    // dragDom.style.left = `${(document.body.clientWidth - width) / 2}px`;
    // 鼠标按下事件
    drawerHeaderEl.onmousedown = (e) => {
      // 鼠标按下，计算当前元素距离可视区的距离 (鼠标点击位置距离可视窗口的距离)
      const disX = e.clientX - drawerHeaderEl.offsetLeft;
      const disY = e.clientY - drawerHeaderEl.offsetTop;

      // 获取到的值带px 正则匹配替换
      let styL, styT;

      // 注意在ie中 第一次获取到的值为组件自带50% 移动之后赋值为px
      if (sty.left.includes('%')) {
        styL = +document.body.clientWidth * (+sty.left.replace(/%/g, '') / 100);
        styT = +document.body.clientHeight * (+sty.top.replace(/%/g, '') / 100);
      } else {
        styL = +sty.left.replace(/\px/g, '');
        styT = +sty.top.replace(/\px/g, '');
      }

      // 鼠标拖拽事件
      document.onmousemove = function (e) {
        // 通过事件委托，计算移动的距离 （开始拖拽至结束拖拽的距离）
        const l = e.clientX - disX;
        const t = e.clientY - disY;

        let finallyL = l + styL
        let finallyT = t + styT

        // 移动当前元素
        dragDom.style.left = `${finallyL}px`;
        dragDom.style.top = `${finallyT}px`;

      };

      document.onmouseup = function (e) {
        document.onmousemove = null;
        document.onmouseup = null;
      };
    }
  }
};
